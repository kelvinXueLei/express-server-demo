

配置
```

PORT=
HOST=

MYSQL_POOL_LIMIT=       mysql 连接池数量
MYSQL_HOST=             mysql host
MYSQL_USER=             mysql username
MYSQL_PASSWORD=         mysql password
MYSQL_DB=               mysql database


LOGGER_TYPE=            日志类型 （参考 log4js）
LOGGER_FILE_PATH=       日志文件路径
LOGGER_APPENDER=        日志 附加字段
LOGGER_LEVEL=           日志级别   trace, debug, info, warn, error, fatal
LOGGER_CONSOLE=         日志是否在控制台打印


```
